  1. Modelltest für Order Effekt.

    Wir testen drei Modelle
    A Basismodell ohne Order
    B plus Order Haupteffekt
    C plus Interaktion planned_condition mal order_group
    –> Wenn C gegen B nicht besser wird, gibt es keinen Hinweis, dass der Condition Effekt von der Reihenfolge abhaengt.

m_base <- lmer(
  total_kcal ~ planned_condition + (1 | subject),
  data = dat_main_order,
  REML = FALSE
)

m_order_add <- lmer(
  total_kcal ~ planned_condition + order_group + (1 | subject),
  data = dat_main_order,
  REML = FALSE
)

m_order_int <- lmer(
  total_kcal ~ planned_condition * order_group + (1 | subject),
  data = dat_main_order,
  REML = FALSE
)

# ============================================================
# Funktionen: report ready Output fuer Mixed Models

fit_main_effect <- function(dat, outcome, use_random_slope = FALSE) {

  if (use_random_slope) {
    f_m1 <- as.formula(
      paste0(outcome, " ~ planned_condition + order_group + (1 + planned_condition || subject)")
    )
  } else {
    f_m1 <- as.formula(
      paste0(outcome, " ~ planned_condition + order_group + (1 | subject)")
    )
  }

  f_m0 <- as.formula(
    paste0(outcome, " ~ order_group + (1 | subject)")
  )

  m1 <- lmer(f_m1, data = dat, REML = FALSE)
  m0 <- lmer(f_m0, data = dat, REML = FALSE)

  list(m0 = m0, m1 = m1)
}
model_summary_pack <- function(m0, m1, dat, outcome) {

  comp <- anova(m0, m1)
  ci <- confint(m1, method = "Wald")

  beta <- fixef(m1)
  beta_cond <- beta[grep("^planned_condition", names(beta))][1]

  sd_y <- dat |>
    summarise(sd_y = sd(.data[[outcome]], na.rm = TRUE)) |>
    pull(sd_y)

  eff_sd <- beta_cond / sd_y

  list(
    comparison = comp,
    fixed = summary(m1)$coefficients,
    ci = ci,
    eff_sd = eff_sd
  )
}

# ------------------------------------------------------------
# Mini QC Funktion: wie viele vollständige Faelle hat ein Moderationssample
# ------------------------------------------------------------
qc_mod_sample <- function(dat, outcome, cond = "planned_condition", mod) {
  dat |>
    summarise(
      n_rows = n(),
      n_ids = n_distinct(fastress_id),
      n_nonmiss_outcome = sum(!is.na(.data[[outcome]])),
      n_nonmiss_cond = sum(!is.na(.data[[cond]])),
      n_nonmiss_mod = sum(!is.na(.data[[mod]])),
      n_complete_for_model = sum(!is.na(.data[[outcome]]) & !is.na(.data[[cond]]) & !is.na(.data[[mod]]))
    )
}

# --------------------------------------------------
# Guard: stop if no complete cases for the moderation model
# --------------------------------------------------
stop_if_no_cases <- function(qc_tbl) {
  n_cc <- qc_tbl$n_complete_for_model[1]
  if (is.na(n_cc) || n_cc == 0) {
    stop("Keine vollstaendigen Faelle fuer dieses Moderationsmodell. Sample ist leer oder hat NAs.")
  }
}
