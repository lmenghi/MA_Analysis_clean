# ============================================================
# Define metabolic condition per session based on blood markers
# ============================================================

blood_condition <- blood_all |>
  mutate(
    fasting_ketones = (
      (ketones_pre >= 0.5 & ketones_pre <= 1.5) |
      (ketones_post >= 0.5 & ketones_post <= 1.5)
    ),
    fasting_glucose = (
      (glucose_pre >= 3.9 & glucose_pre <= 5.6) |
      (glucose_post >= 3.9 & glucose_post <= 5.6)
    ),
    satiated_ketones = (
      ketones_pre < 0.6 |
      ketones_post < 0.6
    ),
    satiated_glucose = (
      (glucose_pre >= 6.7 & glucose_pre <= 7.8) |
      (glucose_post >= 6.7 & glucose_post <= 7.8)
    ),
    condition = case_when(
      fasting_ketones | fasting_glucose ~ "fasted",
      satiated_ketones | satiated_glucose ~ "satiated",
      TRUE ~ NA_character_
    )
  ) |>
  select(fastress_id, session, condition)

row_mean_safe <- function(df, items, min_n = 3) {
  items <- intersect(items, names(df))
  if (length(items) == 0) return(rep(NA_real_, nrow(df)))

  m <- df |>
    select(all_of(items)) |>
    mutate(across(everything(), ~ readr::parse_number(str_replace(as.character(.x), ",", ".")))) |>
    as.matrix()

  n_nonmiss <- rowSums(!is.na(m))
  out <- rowMeans(m, na.rm = TRUE)
  out[n_nonmiss < min_n] <- NA_real_
  out
}
